<script>

// Инициализация Telegram Web App

const tg = window.Telegram.WebApp;

tg.expand(); // Растянуть на весь экран

// Обработчик переключения страниц

document.querySelectorAll('.nav-button').forEach(button => {

button.addEventListener('click', () => {

const pageNumber = button.dataset.page;

// Убрать активные классы

document.querySelectorAll('.page, .nav-button').forEach(el => {

el.classList.remove('active');

});

// Показать выбранную страницу

document.getElementById(`page${pageNumber}`).classList.add('active');

button.classList.add('active');

});

});



// Инициализация основной кнопки Telegram

tg.MainButton.setText("Готово");

tg.MainButton.show();

// Обработчик основной кнопки

tg.MainButton.onClick(() => {

tg.close();

});
